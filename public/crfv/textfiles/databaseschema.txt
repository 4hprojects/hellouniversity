CREATE TABLE public.attendance_records (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  event_id text NOT NULL,
  status text,
  attendee_id text,
  note text,
  rfid text,
  slot text,
  time time without time zone,
  date date,
  old_event_id text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  timestamp timestamp without time zone DEFAULT now(),
  raw_first_name text,
  raw_last_name text,
  raw_rfid text,
  is_unregistered boolean DEFAULT false,
  attendee_no text,
  first_name text,
  last_name text,
  CONSTRAINT attendance_records_pkey PRIMARY KEY (id),
  CONSTRAINT attendance_records_attendee_id_fkey FOREIGN KEY (attendee_id) REFERENCES public.attendees(id),
  CONSTRAINT attendance_records_event_id_fkey FOREIGN KEY (event_id) REFERENCES public.events(event_id)
);

  accommodation text,
  event_id text,
